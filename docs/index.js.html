    stock = 10;

    /**
     * Constructor de la clase artículo
     * @param {String} titulo 
     * @param {Number} precio 
     */
    constructor(titulo, precio) {
        this.titulo = titulo;
        this.precio = precio;
    }

    /**
     * Función que devuelve el stock
     * @returns {number}
     */
    getStock() {
        return this.stock;
    }
}
